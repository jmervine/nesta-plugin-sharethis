def sharethis_buttons
  buttons = []
  STDOUT.puts Nesta::Config.sharethis_buttons.inspect
  Nesta::Config.sharethis_buttons.each do |button|
    buttons << Nesta::Plugin::Sharethis::BUTTONS[button]
  end
  buttons.join("\n")
end
